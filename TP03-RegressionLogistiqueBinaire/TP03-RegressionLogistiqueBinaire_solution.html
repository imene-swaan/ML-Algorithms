def normaliser(X, mean=None, std=None):

# th0 + th1 * x1 + th2 * x2 = 0
def ligne_decision(X, Theta, mean=[0, 0], std=[1, 1]):
    X1 = [np.min(X[:,0]), np.max(X[:,0])]
    X2_min = mean[1] - std[1] * (Theta[0] + Theta[1] * (X1[0] - mean[0])/std[0]) / Theta[2]
    X2_max = mean[1] - std[1] * (Theta[0] + Theta[1] * (X1[1] - mean[0])/std[0]) / Theta[2]
    return X1, [X2_min, X2_max]

1.3- La fonction du cout¶

$$ cout(h_\theta(x), y) = \begin{cases} - \log(h_\theta(x)) & \text{ si } y = 1\\ - \log(1 - h_\theta(x)) & \text{ si } y = 0 \end{cases}$$

Puisque $ y \in \{0, 1\}$ donc,

$$ cout(h_\theta(x), y) = - y \log(h_\theta(x)) - (1-y) \log(1 - h_\theta(x))$$$$J(\theta) = \frac{1}{m} \sum\limits_{i=1}^{m} cout(h_\theta(x^{(i)}), y^{(i)})$$

1.5- Entraînnement (algorithme du gradient)¶

Les cofficients sont mis jour itrativement en se basant sur le gradient et un pas d'apprentissage $\alpha$. Puisque cette fonction a t implmente dans le TP prcdent, elle est donne ici, mais d'une faon plus parametrable.

$$\theta_j = \theta_j - \alpha \frac{\partial J(\theta)}{\theta_j} $$

Notre fonction d'entrainnement prend les paramtres suivantes :

  • X : matrice (chantillons X caractristiques)
  • Y : vecteur (ou matrice) des rsultas (chantillons X nombre_classes)
  • norm : si on normalise X ou nom (par dfaut : True)
  • const : si on ajoute $\theta_0$ ou non (par dfaut : True)
  • nbr_iter : nombre des itrations avant de sortir
  • alpha : le pas d'apparentissage (Learning rate)
  • eps : le test d'arrt si la diffrence entre les couts (actuel et prcdent) est infrieur $\epsilon$ on arrte la dsente mme si on n'a pas termin toutes les itrations
  • theta_func : la fonction qui gnre les $\theta$ (par dfaut : zros)
  • h_func : la fonction qui calcule les probabilits (par dfaut : h_1)
  • J_func : la fonction du cout (par dfaut : J_1)
  • grad_func : la fonction qui calcule le gradient (par dfaut : gradient_1)

def entrainer_1(X, Y, norm=True, const=True, nbr_iter=200, alpha=1., eps=0.01, 
                theta_func=generer_zeros_1, h_func=h_1, J_func=J_1, grad_func=gradient_1): 
    
    X_pre, mean, std = preparer(X, norm, const)
    Theta = theta_func(X_pre.shape[1])
    
    couts = []
    couts.append(J_func(h_func(X_pre, Theta), Y))
    
    for i in range(nbr_iter):
        H = h_func(X_pre, Theta)
        Theta -= alpha * grad_func(X_pre, H, Y)
        couts.append(J_func(H, Y))
    
    return Theta, mean, std, couts

1.7- Regrouper les fonctions ensemble¶

Pour bien grer l'entrainnement et la prdiction, on rassemble les fonctions que vous avez implment dans une seul classe. L'intrt :

  • Si on applique la normalisation durant l'entrainnement, on doit l'appliquer aussi durant la prdiction. En plus, on doit utiliser les mmes paramtres (moyenne et cart-type)
  • On utilise les thtas optimales lors de la prdicition

class RegLogistique(object):
    
    def __init__(self, nbr_iter=100, alpha=1., theta_func=generer_zeros_1, norm=True, const=True): 
        self.nbr_iter = nbr_iter
        self.alpha = alpha
        self.theta_func = theta_func
        self.norm = norm
        self.const = const
    
    def entrainer(self, X, Y): 
        self.Theta, self.mean, self.std, self.couts = entrainer_1(X, Y, 
                                                                  nbr_iter=self.nbr_iter, 
                                                                  alpha=self.alpha, 
                                                                  theta_func=self.theta_func, 
                                                                  norm=self.norm, 
                                                                  const=self.const)
        
    # La prdiction
    # si prob=True elle rend un vecteur de probabilits
    # sinon elle rend une vecteur de 1 et 0
    def predire(self, X, prob=True, seuil=0.5):
        H = h_1(X, self.Theta, self.mean, self.std, self.const)
        if prob:
            return H
        return predire_1(H, seuil=seuil)

3.5- Test¶

On mesure la qualit d'un systme de classification en utilisant la matrice de confusion :

- Estim (1) Estim (0)

def calculerTP(Y_reel, Y_predit): 
    return (Y_reel * Y_predit).sum()

def calculerFN(Y_reel, Y_predit): 
    return (Y_reel * (1-Y_predit)).sum()

def calculerFP(Y_reel, Y_predit): 
    return ((1-Y_reel) * Y_predit).sum()

def calculerTN(Y_reel, Y_predit): 
    return ((1-Y_reel) * (1-Y_predit)).sum()
